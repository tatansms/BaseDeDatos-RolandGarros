SET datestyle TO 'ISO, DMY';
CREATE TABLE Torneos (
	idTorneo SERIAL PRIMARY KEY,
	fechaInicioDelTorneo DATE NOT NULL,
	fechaCierreDelTorneo DATE NOT NULL
);

CREATE TABLE Noticias(
	idNoticia SERIAL PRIMARY KEY,
	fechaNoticia DATE NOT NULL,
	horaDePublicación TIME NOT NULL,
	tituloNoticia VARCHAR(100) NOT NULL,
	Descripción VARCHAR(200) NULL
);

CREATE TABLE NoticiasDelTorneo(
	idTorneo INT NOT NULL,
	idNoticia INT NOT NULL,
	PRIMARY KEY(idTorneo,idNoticia),
	FOREIGN KEY(idTorneo) REFERENCES Torneos(idTorneo),
	FOREIGN KEY(idNoticia) REFERENCES Noticias(idNoticia)
);
 --                                              INSERCIÓN LISTA
CREATE TABLE Paises(
	idPais SERIAL PRIMARY KEY,
	nombre VARCHAR(50) NOT NULL
);

CREATE TABLE Personas(
	idPersona SERIAL PRIMARY KEY,
	primerNombre VARCHAR(50) NOT NULL,
	segundoNombre VARCHAR(50) NULL,
	primerApellido VARCHAR(50) NOT NULL,
	segundoApellido VARCHAR(100) NULL,
	sexo VARCHAR(1) CHECK(sexo in('M','F')) NOT NULL,
	idPais INT NOT NULL,
	FOREIGN KEY(idPais) REFERENCES Paises(idPais)
);

CREATE TABLE noticiasPorPersona(
	idPersona INT NOT NULL,
	idNoticia INT NOT NULL,
	PRIMARY KEY(idPersona,idNoticia),
	FOREIGN KEY(idPersona) REFERENCES Personas(idPersona),
	FOREIGN KEY(idNoticia) REFERENCES Noticias(idNoticia)
);

CREATE TABLE categoriasDelTorneo(
	idCategoria SERIAL PRIMARY KEY,
	nombre VARCHAR(50) NOT NULL,
	numeroDeParticipantes INT NOT NULL,
	nombreDeLaCopa VARCHAR(50) NOT NULL,
	premioMayor INT NOT NULL,
	descripcion VARCHAR(100) NULL
);

CREATE TABLE Tenistas(
	idTenista SERIAL PRIMARY KEY,
	peso NUMERIC NOT NULL,
	Altura NUMERIC NOT NULL,
	fechaNacimiento DATE NOT NULL,
	puntajeRanking INT NOT NULL,
	idCategoria INT NOT NULL,
	FOREIGN KEY(idCategoria) REFERENCES categoriasDelTorneo(idCategoria),
	FOREIGN KEY(idTenista) REFERENCES Personas(idPersona),
	CONSTRAINT fechas_validas CHECK(fechaNacimiento<CURRENT_DATE)
);

CREATE TABLE Ganadores(
	idGanador SERIAL PRIMARY KEY,
	titulo VARCHAR(100) NOT NULL,
	idTenista INT NOT NULL,
	idCategoriaModalidad INT NOT NULL,
	idTorneo INT NOT NULL,
	FOREIGN KEY(idTenista) REFERENCES Tenistas(idTenista),
	FOREIGN KEY(idCategoriaModalidad) REFERENCES categoriasDelTorneo(idCategoria),
	FOREIGN KEY(idTorneo) REFERENCES Torneos(idTorneo)
);

CREATE TABLE tenistasEnELTorneo(
	idTorneo INT NOT NULL,
	idTenista INT NOT NULL,
	PRIMARY KEY(idTorneo,idTenista),
	FOREIGN KEY(idTorneo) REFERENCES Torneos(idTorneo),
	FOREIGN KEY(idTenista) REFERENCES Tenistas(idTenista)
);
CREATE TABLE categoriasDeLosArbitros(
	idCategoria SERIAL PRIMARY KEY,
	nombre VARCHAR(50) NOT NULL,
	descripcion VARCHAR(200) NULL
);

CREATE TABLE Arbitros(
	idArbitro SERIAL PRIMARY KEY,
	idioma VARCHAR(20) NULL,
	experiencia INT NOT NULL,
	idCategoria INT NOT NULL,
	FOREIGN KEY(idArbitro) REFERENCES Personas(idPersona),
	FOREIGN KEY(idCategoria) REFERENCES categoriasDeLosArbitros(idCategoria)
);


CREATE TABLE trayectoriasProfesionales(
	idTenista SERIAL PRIMARY KEY,
	manoDeJuego VARCHAR(10) NOT NULL,
	fechaDeInicioDeCarrera DATE NOT NULL,
	dineroDelPremio INT NOT NULL,
	partidosPerdidos INT NOT NULL,
	partidosGanados INT NOT NULL,
	entrenador VARCHAR(50) NULL,
	FOREIGN KEY(idTenista) REFERENCES Tenistas(idTenista)
);

CREATE TABLE participacionesEnELTorneo(
	idTenista SERIAL PRIMARY KEY,
	RolandGarrosJugados INT NOT NULL,
	mejorActuacionEnDoblesEnRG VARCHAR(100) NOT NULL,
	mejorActuacionEnSingleEnRG VARCHAR(100) NOT NULL,
	FOREIGN KEY(idTenista) REFERENCES Tenistas(idTenista)
);
CREATE TABLE titulosEFinales(
	idTenista SERIAL PRIMARY KEY,
	numeroDeTitulosSiglesGanados INT NULL,
	numeroDeTitulosDoblesGanados INT NULL,
	mejorRankingEnSingles INT NOT NULL,
	mejorRankingEnDobles INT NOT NULL,
	FOREIGN KEY(idTenista) REFERENCES Tenistas(idTenista)
);


CREATE TABLE categoriasPatrocinadores(
	idCategoria SERIAL PRIMARY KEY,
	nombre VARCHAR(50) NOT NULL,
	descripcion VARCHAR(150) NULL
);

CREATE TABLE Patrocinadores(
	idPatrocinador SERIAL PRIMARY KEY,
	nombre VARCHAR(30) NOT NULL,
	idCategoria INT NOT NULL,
	FOREIGN KEY(idCategoria) REFERENCES categoriasPatrocinadores(idCategoria)
);

CREATE TABLE patrocinadoresPorTorneo(
	idPatrocinador INT NOT NULL,
	idTorneo INT NOT NULL,
	PRIMARY KEY(idPatrocinador,idTorneo),
	FOREIGN KEY(idPatrocinador) REFERENCES Patrocinadores(idPatrocinador),
	FOREIGN KEY(idTorneo) REFERENCES Torneos(idTorneo)
);

	
CREATE TABLE fasesDelTorneo(
	idFaseDelTorneo SERIAL PRIMARY KEY,
	nombre VARCHAR(50) NOT NULL,
	descripción VARCHAR(100) NULL
);

CREATE TABLE Pistas(
	idPista SERIAL PRIMARY KEY,
	nombre VARCHAR(100) NOT NULL,
	descripcion VARCHAR(200) NULL,
	dimensiones VARCHAR(50) NOT NULL,
	capacidad INT NOT NULL,
	superficie VARCHAR(100) NOT NULL
);

CREATE TABLE Partidos(
	idPartido SERIAL PRIMARY KEY,
	fecha DATE NOT NULL,
	hora TIME NOT NULL,
	marcador VARCHAR(100) NOT NULL,
	idTorneo INT NOT NULL,
	idGanador INT NOT NULL,
	idPerdedor INT NOT NULL,
	idModalidad INT NOT NULL,
	idFase INT NOT NULL,
	idPista INT NOT NULL,
	FOREIGN KEY(idGanador) REFERENCES Tenistas(idTenista),
	FOREIGN KEY(idPerdedor) REFERENCES Tenistas(idTenista),
	FOREIGN KEY(idModalidad) REFERENCES categoriasDelTorneo(idCategoria),
	FOREIGN KEY(idFase) REFERENCES fasesDelTorneo(idFaseDelTorneo),
	FOREIGN KEY(idPista) REFERENCES Pistas(idPista),
	FOREIGN KEY(idTorneo) REFERENCES Torneos(idTorneo)
);

CREATE TABLE arbitrosPorPartidos(
	idArbitro INT NOT NULL,
	idPartido INT NOT NULL,
	PRIMARY KEY(idArbitro,idPartido),
	FOREIGN KEY(idArbitro) REFERENCES Arbitros(idArbitro),
	FOREIGN KEY(idPartido) REFERENCES Partidos(idPartido)	
);

CREATE TABLE noticiasDelPartido(
	idNoticia INT NOT NULL,
	idPartido INT NOT NULL,
	PRIMARY KEY(idNoticia,idPartido),
	FOREIGN KEY(idNoticia) REFERENCES Noticias(idNoticia),
	FOREIGN KEY(idPartido) REFERENCES Partidos(idPartido)
);


CREATE TABLE mediosDeTransmision(
	idMedioDeTransmision SERIAL PRIMARY KEY,
	nombre VARCHAR(50) NOT NULL,
	idPais INT NOT NULL,
	FOREIGN KEY(idPais) REFERENCES Paises(idPais)
);

CREATE TABLE canalesDeTelevisiones(
	idCanal SERIAL PRIMARY KEY,
	FOREIGN KEY(idCanal) REFERENCES mediosDeTransmision(idMedioDeTransmision)
);

CREATE TABLE Radios(
	idRadio SERIAL PRIMARY KEY,
	FOREIGN KEY(idRadio) REFERENCES mediosDeTransmision(idMedioDeTransmision)
);

CREATE TABLE transmisionesDelTorneo(
	idTorneo INT NOT NULL,
	idMedioDeTransmision INT NOT NULL,
	PRIMARY KEY(idTorneo,idMedioDeTransmision),
	FOREIGN KEY(idTorneo) REFERENCES Torneos(idTorneo),
 	FOREIGN KEY(idMedioDeTransmision) REFERENCES mediosDeTransmision(idMedioDeTransmision)
);


CREATE TABLE categoriasDelSet(
	idCategoriaSet INT PRIMARY KEY NOT NULL,
	nombre VARCHAR(50) NOT NULL,
	descripcion VARCHAR(100) NULL
);


CREATE TABLE Sets(
	idSet INT PRIMARY KEY NOT NULL,
	idTenista INT NOT NULL,
	idCategoriaSet INT NOT NULL,
	idPartido INT NOT NULL,
	FOREIGN KEY(idTenista) REFERENCES Tenistas(idTenista),
	FOREIGN KEY(idCategoriaSet) REFERENCES categoriasDelSet(idCategoriaSet),
	FOREIGN KEY(idPartido) REFERENCES Partidos(idPartido)
);


CREATE TABLE tiposDeRallys(
	idTipoRally SERIAL PRIMARY KEY,
	rallyAgrupado INT NOT NULL
);

CREATE TABLE analisisDeLosRallys(
	idAnalisis INT PRIMARY KEY NOT NULL,
	idTipoRally INT NOT NULL,
	idSet INT NOT NULL,
	idTenista INT NOT NULL,
	FOREIGN KEY(idTipoRally) REFERENCES tiposDeRallys(idTipoRally),
	FOREIGN KEY(idSet) REFERENCES Sets(idSet),
	FOREIGN KEY(idTenista) REFERENCES Tenistas(idTenista),
	valor INT NOT NULL
);

CREATE TABLE categoriasEstadisticas(
	idCategoriaEstadistica SERIAL PRIMARY KEY,
	nombre Varchar(100) NOT NULL
);

CREATE TABLE tiposDeEstadisticas(
	idTipoEstadistica SERIAL PRIMARY KEY,
	nombre VARCHAR(100) NOT NULL,
	idCategoriaEstadistica INT NOT NULL,
	FOREIGN KEY(idCategoriaEstadistica) REFERENCES categoriasEstadisticas(idCategoriaEstadistica)
);

CREATE TABLE Estadisticas(
	idEstadistica SERIAL PRIMARY KEY,
	idSet INT NOT NULL,
	idTipoEstadistica INT NOT NULL,
	FOREIGN KEY(idSet) REFERENCES Sets(idSet),
	FOREIGN KEY(idTipoEstadistica) REFERENCES tiposDeEstadisticas(idTipoEstadistica)	
);
